   210тЖТ   210тЖТ  if (!dayMonthYear) return 'тАФ';
   211тЖТ   211тЖТ  const d = dayMonthYear instanceof Date ? dayMonthYear : new Date(dayMonthYear);
   212тЖТ   212тЖТ  const day = String(d.getDate()).padStart(2, '0');
   213тЖТ   213тЖТ  const month = String(d.getMonth() + 1).padStart(2, '0');
   214тЖТ   214тЖТ  const year = d.getFullYear();
   215тЖТ   215тЖТ  return `${day} / ${month} / ${year}`;
